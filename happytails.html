SQL Queries

Business Analysis

Five queries designed to answer key business questions about customer behavior, sitter performance, facility revenue, and service demand.

Customer Visit History

JOIN

Returns a full record of each visit including customer name, pet details, service type, stay duration, and final payment, ordered by most recent check-in.

SELECT
    c.customer_fname || ' ' || c.customer_lname AS customer_name,
    p.pet_name,
    p.pet_breed,
    s.service_type,
    v.visit_check_in_date,
    v.visit_check_out_date,
    v.visit_days_stayed,
    pay.payment_final_price
FROM customer c
JOIN pet p ON c.customer_id = p.customer_id
JOIN visit v ON p.pet_id = v.pet_id
JOIN service s ON v.service_id = s.service_id
JOIN payment pay ON v.visit_id = pay.visit_id
ORDER BY v.visit_check_in_date DESC;

Sitter Performance by Revenue

GROUP BY

Ranks sitters by total revenue generated, showing visit count and average payment per visit alongside facility assignment.

SELECT
    sit.sitter_fname || ' ' || sit.sitter_lname AS sitter_name,
    f.facility_name,
    COUNT(DISTINCT a.visit_id) AS total_visits,
    SUM(pay.payment_final_price) AS total_revenue,
    AVG(pay.payment_final_price) AS avg_payment
FROM sitter sit
JOIN assignment a ON sit.sitter_id = a.sitter_id
JOIN visit v ON a.visit_id = v.visit_id
JOIN payment pay ON v.visit_id = pay.visit_id
JOIN facility f ON sit.sitter_facility_id = f.facility_id
GROUP BY sit.sitter_id, f.facility_name
ORDER BY total_revenue DESC;

Facility Performance Summary

AGGREGATION

Compares facilities by total visits, unique pets served, revenue, and average stay duration to identify top-performing locations.

SELECT
    f.facility_name,
    f.facility_state,
    COUNT(DISTINCT v.visit_id) AS total_visits,
    COUNT(DISTINCT p.pet_id) AS unique_pets,
    SUM(pay.payment_final_price) AS total_revenue,
    AVG(v.visit_days_stayed) AS avg_stay_duration
FROM facility f
JOIN visit v ON f.facility_id = v.facility_id
JOIN pet p ON v.pet_id = p.pet_id
JOIN payment pay ON v.visit_id = pay.visit_id
GROUP BY f.facility_id
ORDER BY total_revenue DESC;

Customer Type Analysis

GROUP BY

Segments customers by type to compare visit frequency, total spend, average payment per visit, and loyalty points earned.

SELECT
    c.customer_type,
    COUNT(DISTINCT c.customer_id) AS customer_count,
    COUNT(v.visit_id) AS total_visits,
    SUM(pay.payment_final_price) AS total_spent,
    AVG(pay.payment_final_price) AS avg_per_visit,
    SUM(pay.payment_loyalty_points) AS total_loyalty_points
FROM customer c
JOIN pet p ON c.customer_id = p.customer_id
JOIN visit v ON p.pet_id = v.pet_id
JOIN payment pay ON v.visit_id = pay.visit_id
GROUP BY c.customer_type;

Service Demand and Promo Usage

CASE WHEN

Ranks services by total revenue, comparing base price to actual average price and tracking promotional code usage per service type.

SELECT
    s.service_type,
    COUNT(v.visit_id) AS bookings,
    s.service_base_price,
    SUM(pay.payment_final_price) AS total_revenue,
    AVG(pay.payment_final_price) AS avg_actual_price,
    COUNT(CASE WHEN pay.payment_promo_code IS NOT NULL THEN 1 END) AS promo_usage_count
FROM service s
JOIN visit v ON s.service_id = v.service_id
JOIN payment pay ON v.visit_id = pay.visit_id
GROUP BY s.service_id
ORDER BY total_revenue DESC;
